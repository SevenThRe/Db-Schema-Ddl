**SQL parsing for live execution:**
Reuse the EXISTING server-side preview mechanism. The current component already calls `usePreviewDdlImport()` which returns parsed table definitions. For live execution, the raw SQL text (from the existing `sqlText` state or file content) must be split into statements.

Use a simple semicolon-based splitter for the live-execution tab (the server-side DDL parser handles CREATE TABLE specifically; live execution needs arbitrary SQL statement splitting):
```typescript
function splitSqlStatements(sql: string): Array<{ sql: string; lineNumber: number }> {
  const statements: Array<{ sql: string; lineNumber: number }> = [];
  let current = "";
  let lineNum = 1;
  let stmtStartLine = 1;
  for (const line of sql.split("\n")) {
    const trimmed = line.trim();
    if (trimmed.startsWith("--") || trimmed === "") {
      lineNum++;
      if (!current.trim()) stmtStartLine = lineNum;
      continue;
    }
    current += line + "\n";
    if (trimmed.endsWith(";")) {
      const stmt = current.trim().replace(/;$/, "").trim();
      if (stmt) statements.push({ sql: stmt, lineNumber: stmtStartLine });
      current = "";
      stmtStartLine = lineNum + 1;
    }
    lineNum++;
  }
  const remaining = current.trim().replace(/;$/, "").trim();
  if (remaining) statements.push({ sql: remaining, lineNumber: stmtStartLine });
  return statements;
}
```

```typescript
async function executeLiveImport() {
  if (!selectedConnectionId || liveStatements.length === 0) return;
  setIsExecuting(true);

  for (let i = 0; i < liveStatements.length; i++) {
    const stmt = liveStatements[i];
    setLiveStatements(prev => prev.map((s, idx) => idx === i ? { ...s, status: "running" } : s));

    try {
      // 1. 危険な SQL を事前チェック
      const preview = await hostApi.connections.previewDangerousSql(selectedConnectionId, stmt.sql);

      let confirmed = false;
      if (preview.dangers && preview.dangers.length > 0) {
        // 2. DangerousSqlDialog を表示して確認を待つ
        confirmed = await new Promise<boolean>((resolve) => {
          setDangerPreview(preview);
          setPendingConfirmIndex(i);
          // resolve は DangerousSqlDialog の onConfirm / onCancel で呼ばれる
          dangerResolveRef.current = resolve;
        });
        setDangerPreview(null);
        setPendingConfirmIndex(null);

        if (!confirmed) {
          setLiveStatements(prev => prev.map((s, idx) => idx === i ? { ...s, status: "skipped" } : s));
          continue;
        }
      }

      // 3. executeQuery を呼び出す
      const result = await hostApi.connections.executeQuery({
        connectionId: selectedConnectionId,
        sql: stmt.sql,
        requestId: `ddl-import-${Date.now()}-${i}`,
        confirmed: preview.dangers && preview.dangers.length > 0,
        maxRows: 0,
      });

      if (result.error) {
        setLiveStatements(prev => prev.map((s, idx) => idx === i ? {
          ...s, status: "error", error: result.error, elapsedMs: result.elapsedMs,
        } : s));
      } else {
        setLiveStatements(prev => prev.map((s, idx) => idx === i ? {
          ...s, status: "success", elapsedMs: result.elapsedMs,
        } : s));
      }
    } catch (err) {
      setLiveStatements(prev => prev.map((s, idx) => idx === i ? {
        ...s, status: "error", error: String(err),
      } : s));
    }
  }
  setIsExecuting(false);
}
```

const liveFileInputRef = useRef<HTMLInputElement>(null);

function handleLoadSqlFile() {

async function handleLiveFileChange(e: ChangeEvent<HTMLInputElement>) {
  const file = e.target.files?.[0];
  if (!file) return;
  const text = await file.text();
  const statements = splitSqlStatements(text);
  setLiveStatements(statements.map((s, i) => ({
    index: i,
    sql: s.sql,
    lineNumber: s.lineNumber,
    status: "pending" as const,
  })));
  e.target.value = "";
}
